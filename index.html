<script>
let slides=document.querySelectorAll(".slide");
let index=0;

/* AUTO SLIDE */
setInterval(()=>{
  slides[index].classList.remove("active");
  index=(index+1)%slides.length;
  slides[index].classList.add("active");
},6000);

/* COUNTDOWN */
let target=new Date("May 2, 2026 00:00:00").getTime();

setInterval(()=>{
  let now=new Date().getTime();
  let diff=target-now;

  let d=Math.floor(diff/(1000*60*60*24));
  let h=Math.floor((diff%(1000*60*60*24))/(1000*60*60));
  let m=Math.floor((diff%(1000*60*60))/(1000*60));

  document.getElementById("countdown").innerHTML =
  d+" hari "+h+" jam "+m+" menit lagi ❤️";
},1000);

/* PARTICLES */
for(let i=0;i<40;i++){
  let p=document.createElement("div");
  p.classList.add("particle");

  let size=Math.random()*6+4;
  p.style.width=size+"px";
  p.style.height=size+"px";
  p.style.left=Math.random()*100+"vw";
  p.style.background=Math.random()>0.5?"pink":"white";
  p.style.animationDuration=(Math.random()*8+6)+"s";

  document.body.appendChild(p);
}

/* ENDING */
let played=false;

function playEnding(){
  if(played) return;
  played=true;

  let text="Maaf aku belum bisa selalu ada di samping kamu...";
  let el=document.getElementById("endingText");

  let i=0;
  function typing(){
    if(i<text.length){
      el.innerHTML+=text.charAt(i);
      i++;
      setTimeout(typing,60);
    }
  }
  typing();

  setTimeout(()=>{
    document.getElementById("endingSub").style.opacity=1;
  },4000);

  setTimeout(()=>{
    document.getElementById("voice").play();
  },5000);
}

setInterval(()=>{
  if(slides[index].id==="ending"){
    playEnding();
  }
},1000);
</script>
